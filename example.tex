\defbeamertemplate*{title page}{myTitle}[1][]
{
	\vfill
	\begingroup
		\centering
			\begin{beamercolorbox}[sep=8pt,center,#1]{title}
			\usebeamerfont{title}\inserttitle\par%
			\ifx\insertsubtitle\@empty\relax%
			\else%
			\vskip0.25em%
			{\usebeamerfont{subtitle}\usebeamercolor[fg]{subtitle}\insertsubtitle\par}%
			\fi%
		\end{beamercolorbox}%
		\vskip1em\par

		\begin{columns}

		\hfill
		\begin{column}{0.5\canvaswidth} %
			\begin{beamercolorbox}[sep=8pt,center,#1]{author}
				\usebeamerfont{author}\textcolor{black}{\insertauthorname, \authortitle}
				\usebeamerfont{institute}\\ \insertauthoremail \\ \authorphone
			\end{beamercolorbox}
			\begin{beamercolorbox}[sep=8pt,center,#1]{institute}
				\usebeamerfont{institute}\insertinstitute
			\end{beamercolorbox}
			\begin{beamercolorbox}[sep=8pt,center,#1]{date}
				\usebeamerfont{date}\insertdate
			\end{beamercolorbox}\vskip0.5em
			{\usebeamercolor[fg]{titlegraphic}\inserttitlegraphic\par}
		\end{column}
		\hfill
		\begin{column}{0.5\canvaswidth} %
			\begin{scriptsize}
			\lipsum[4]
			\end{scriptsize}
		\end{column}
		\hfill
		\end{columns}
	\endgroup
	\vfill
}

\newcommand<>{\uncovergraphics}[2][{}]{
    % SEE: https://tex.stackexchange.com/a/354033/95423
    \begin{tikzpicture}
    \node[anchor=south west,inner sep=0] (B) at (4,0)
        {\includegraphics[#1]{#2}};
    \alt#3{}{%
        \fill [draw=none, fill=background, fill opacity=0.9] (B.north west) -- (B.north east) -- (B.south east) -- (B.south west) -- (B.north west) -- cycle;
    }
    \end{tikzpicture}
}
